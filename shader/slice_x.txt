float offset = m_count;

float map(vec3 v) {
	float dx = v.x + offset;

	if (v.x >= m_offset) {
		if (dx >= i_volume_size.x) {
			return 0.0;
		} else {
			return voxel(vec3(dx, v.y, v.z));
		}
	} else {
		return voxel(v);
	}
}
