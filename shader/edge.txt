bool isSelectedPal(float p) {
	for (int i = 0; i < i_num_color_sels; i += 1) {
		if (p == color_sel(float(i))) {
			return true;
		}
	}
	return false;
}

float map(vec3 v) {
	float index = voxel(v);
	float adj = 0.0;

	if (isSelectedPal(index)) {
		for (float x = -1.0; x <= 1.0; x += 1.0) {
			for (float y = -1.0; y <= 1.0; y += 1.0) {
				for (float z = -1.0; z <= 1.0; z += 1.0) {
					vec3 p = v + vec3(x, y, z);

					if (voxel(p) == 0.0) {
						adj += 1.0;
					}
				}
			}
		}

		if (adj >= 10.0) {
			return m_color;
		}
	}


	return index;
}
