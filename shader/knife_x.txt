bool isSelectedPal(float p) {
	for (int i = 0; i < i_num_color_sels; i += 1) {
		if (p == color_sel(float(i))) {
			return true;
		}
	}
	return false;
}

float map(vec3 v) {
	float a[256];
	int count = 0;

	for (float x = 0.0; x <= i_volume_size.x; x += 1.0) {
		float k = voxel(vec3(x, v.yz));

		if (isSelectedPal(k) == false) {
			a[count] = k;
			count += 1;
		}
	}

	if (v.x >= float(count)) {
		return 0.0;
	} else {
		int n = int(v.x);
		return a[n];
	}
}
